   ```rust
   #[command(name = "sets")]
   Sets {
       #[command(subcommand)]
       command: ProfileSetCommand,
   },
   ```

The plural "sets" avoids all conflicts with the existing `Set` (singular) that does config value setting.   
   ```rust
   // Pre-slice compatibility warnings
   if let (Some(ref machine_name), Some(ref filament_name)) = (&options.machine, &options.filament) {
       if let Ok(Some(index)) = slicecore_engine::profile_library::ProfileIndex::load(
           &slicecore_engine::profile_library::default_index_path(options.profiles_dir.as_deref())
       ) {
           let machine_entries: Vec<&ProfileIndexEntry> = index.profiles.iter()
               .filter(|e| e.profile_type == "machine" && e.id.contains(machine_name))
               .collect();
           let filament_entry = index.profiles.iter()
               .find(|e| e.profile_type == "filament" && (e.id.contains(filament_name) || e.name.contains(filament_name)));

           if let Some(filament) = filament_entry {
               use slicecore_engine::enabled_profiles::CompatibilityInfo;
               let report = CompatibilityInfo::compat_report(filament, &machine_entries, 300.0);
               for warning in report.warnings() {
                   match warning {
                       slicecore_engine::enabled_profiles::CompatCheck::NozzleMismatch { profile_nozzle, printer_nozzles } => {
                           eprintln!("Warning: Filament profile specifies {:.1}mm nozzle, but printer supports {:?}mm", profile_nozzle, printer_nozzles);
                       }
                       slicecore_engine::enabled_profiles::CompatCheck::TemperatureWarning { filament_min, printer_max } => {
                           eprintln!("Warning: Filament requires {:.0}C minimum, printer max is {:.0}C", filament_min, printer_max);
                       }
                       _ => {}
                   }
               }
           }
       }
   }
   ```
