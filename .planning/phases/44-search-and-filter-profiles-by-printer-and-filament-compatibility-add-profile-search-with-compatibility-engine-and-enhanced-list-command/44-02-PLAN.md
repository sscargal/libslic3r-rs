1. Add a clap-compatible `CliProfileFilters` struct at the top of `profile_command.rs` that mirrors the engine `ProfileFilters` but with clap derives:
   ```rust
   /// Shared filter flags for profile search and list commands.
   #[derive(clap::Args, Debug, Default, Clone)]
   pub struct CliProfileFilters {
       /// Filter by material type (PLA, ABS, PETG, etc.)
       #[arg(short = 'm', long)]
       pub material: Option<String>,

       /// Filter by vendor name
       #[arg(short = 'v', long)]
       pub vendor: Option<String>,

       /// Filter by nozzle diameter (mm)
       #[arg(short = 'n', long)]
       pub nozzle: Option<f64>,

       /// Filter by profile type (machine, filament, process)
       #[arg(short = 't', long = "type")]
       pub profile_type: Option<String>,
   }
   ```

   Add a conversion method:
   ```rust
   impl From<&CliProfileFilters> for slicecore_engine::profile_library::ProfileFilters {
       fn from(cli: &CliProfileFilters) -> Self {
           Self {
               material: cli.material.clone(),
               vendor: cli.vendor.clone(),
               nozzle: cli.nozzle,
               profile_type: cli.profile_type.clone(),
           }
       }
   }
   ```

2. Update the `Search` variant in `ProfileCommand`:
   ```rust
   Search {
       /// Search query (matches name, vendor, material, ID)
       query: String,

       /// Maximum number of results
       #[arg(short, long, default_value = "20")]
       limit: usize,

       #[command(flatten)]
       filters: CliProfileFilters,

       /// Show only profiles compatible with enabled printers (default)
       /// Use --include-incompatible to show all
       #[arg(long)]
       include_incompatible: bool,

       /// Enable matched profiles after search
       #[arg(long)]
       enable: bool,

       /// Show all profiles regardless of activation status
       #[arg(long)]
       all: bool,

       /// Override profiles directory
       #[arg(long)]
       profiles_dir: Option<PathBuf>,

       /// Output as JSON
       #[arg(long)]
       json: bool,
   },
   ```

3. Update the `List` variant -- replace the existing individual `vendor`, `material` args with the flattened `CliProfileFilters`, and add `--compat` flag:
   ```rust
   List {
       #[command(flatten)]
       filters: CliProfileFilters,

       /// Filter by nozzle size (kept for backward compat, prefer --nozzle in filters)
       #[arg(long)]

       /// Override profiles directory

       /// Output as JSON

       json: bool,

       /// Show only enabled profiles (default)
       #[arg(long)]
       enabled: bool,

       /// Show only disabled profiles
       #[arg(long)]
       disabled: bool,

       /// Show all profiles regardless of activation status
       #[arg(long, conflicts_with_all = ["enabled", "disabled"])]
       all: bool,

       /// Show compatibility column
       #[arg(long)]
       compat: bool,
   },
   ```
   NOTE: The existing `List` has `vendor`, `profile_type`, `material` as separate `#[arg(long)]` fields. Remove those individual fields and replace with the flattened `CliProfileFilters`. The `--vendor` and `--material` flags keep the same long names through CliProfileFilters. The `--profile-type` becomes `--type` (short `-t`). This is a breaking change in flag naming for `--profile-type` -> `--type`, but it aligns search and list.
