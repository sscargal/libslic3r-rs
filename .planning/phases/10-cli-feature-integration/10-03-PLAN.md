        model = \"gpt-4o\"
        api_key = \"sk-...\"

        # Anthropic:
        provider = \"anthropic\"
        model = \"claude-sonnet-4-20250514\"
        api_key = \"sk-ant-...\"

      Then pass it with: slicecore ai-suggest model.stl --ai-config provider.toml"
