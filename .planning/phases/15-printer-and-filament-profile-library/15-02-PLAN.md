1. Add a `find_profiles_dir` helper function to the CLI:
   ```rust
   fn find_profiles_dir(cli_override: Option<&std::path::Path>) -> Option<PathBuf> {
       // 1. CLI flag override
       if let Some(dir) = cli_override {
           return Some(dir.to_path_buf());
       }
       // 2. Environment variable
       if let Ok(dir) = std::env::var("SLICECORE_PROFILES_DIR") {
           let p = PathBuf::from(dir);
           if p.exists() { return Some(p); }
       }
       // 3. Relative to binary (for installed location)
       if let Ok(exe) = std::env::current_exe() {
           if let Some(parent) = exe.parent() {
               let profiles = parent.join("profiles");
               if profiles.exists() { return Some(profiles); }
               // For cargo run: target/debug/slicecore -> ../../profiles
               if let Some(gp) = parent.parent().and_then(|p| p.parent()).and_then(|p| p.parent()) {
                   let profiles = gp.join("profiles");
                   if profiles.exists() { return Some(profiles); }
               }
           }
       }
       // 4. Current directory
       let cwd_profiles = PathBuf::from("profiles");
       if cwd_profiles.exists() { return Some(cwd_profiles); }
       None
   }
   ```

2. Add three new subcommand variants to the `Commands` enum:

   ```rust
   /// List profiles from the profile library.
   ListProfiles {
       /// Filter by vendor name (e.g., BBL, Creality, Prusa).
       #[arg(long)]
       vendor: Option<String>,

       /// Filter by profile type (filament, process, machine).
       #[arg(long, value_name = "TYPE")]
       profile_type: Option<String>,

       /// Filter by material type (PLA, ABS, PETG, TPU, etc.).
       #[arg(long)]
       material: Option<String>,

       /// List available vendors only (no individual profiles).
       #[arg(long)]
       vendors: bool,

       /// Path to profiles directory (overrides auto-detection).
       #[arg(long)]
       profiles_dir: Option<PathBuf>,

       /// Output as JSON instead of human-readable table.
       #[arg(long)]
       json: bool,
   },

   /// Search profiles by keyword (matches name, vendor, material, printer model).
   SearchProfiles {
       /// Search query (case-insensitive substring match across all fields).
       query: String,

       /// Maximum results to show (default: 20).
       #[arg(short, long, default_value = "20")]
       limit: usize,

       /// Path to profiles directory (overrides auto-detection).
       #[arg(long)]
       profiles_dir: Option<PathBuf>,

       /// Output as JSON.
       #[arg(long)]
       json: bool,
   },

   /// Show details of a specific profile from the library.
   ShowProfile {
       /// Profile ID (e.g., orcaslicer/BBL/filament/Bambu_PLA_Basic_BBL_A1).
       id: String,

       /// Show the full TOML content instead of metadata summary.
       #[arg(long)]
       raw: bool,

       /// Path to profiles directory (overrides auto-detection).
       #[arg(long)]
       profiles_dir: Option<PathBuf>,
   },
   ```

3. Implement `cmd_list_profiles` handler:
   - Call `find_profiles_dir(profiles_dir.as_deref())` -- exit with error if None
   - Call `load_index(&profiles_dir)` to load index.json
   - If `--vendors` flag: collect unique vendor names from index, sort, print one per line. If `--json`, emit as JSON array.
   - Otherwise: filter `index.profiles` by vendor (case-insensitive contains), profile_type (exact match), and material (case-insensitive contains)
   - Print results in a tabular format: `TYPE  VENDOR  NAME  MATERIAL  PATH`
   - If `--json`, serialize filtered entries as JSON array to stdout

4. Implement `cmd_search_profiles` handler:
   - Load index via `find_profiles_dir` + `load_index`
   - Split query into whitespace-separated terms (lowercase)
   - Filter profiles where ALL terms match at least one of: name, vendor, material, printer_model, profile_type, quality (all lowercased)
   - Truncate results to `limit`
   - Print results in tabular format or JSON

5. Implement `cmd_show_profile` handler:
   - Load index via `find_profiles_dir` + `load_index`
   - Find entry where `entry.id == id` (exact match) or `entry.path.trim_end_matches(".toml") == id`
   - If not found, try case-insensitive match and suggest correction
   - If `--raw`: read the TOML file at `profiles_dir.join(&entry.path)` and print to stdout
   - If not `--raw`: print structured metadata summary:
     ```
     Profile: Bambu PLA Basic @BBL A1
     Source:  orcaslicer
     Vendor:  BBL
     Type:    filament
     Material: PLA
     Path:    orcaslicer/BBL/filament/Bambu_PLA_Basic_BBL_A1.toml
     ```

6. Wire all three subcommands in the main match block.

7. Update the CLI `after_help` text to add profile discovery commands documentation:
