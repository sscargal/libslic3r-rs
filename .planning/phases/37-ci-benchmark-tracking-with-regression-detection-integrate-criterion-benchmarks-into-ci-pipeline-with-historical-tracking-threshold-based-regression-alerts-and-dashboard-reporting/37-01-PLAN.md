Create `scripts/bench-with-memory.sh` with the following exact content:

```bash
#!/bin/bash
# Runs a single criterion benchmark with /usr/bin/time -v to capture peak RSS.
# Usage: ./scripts/bench-with-memory.sh <package> <bench-name> [extra-cargo-args...]
# Output: Appends bencher-format timing to bench-results/output.txt
#         Appends "MEMORY:<bench-name>:<peak_rss_kb>kB" to bench-results/memory.txt
set -euo pipefail

PACKAGE="$1"
BENCH_NAME="$2"
shift 2

mkdir -p bench-results

TIME_OUTPUT=$(mktemp)

# Run bench: stdout = bencher format timing, stderr = /usr/bin/time stats
/usr/bin/time -v cargo bench -p "$PACKAGE" --bench "$BENCH_NAME" "$@" \
  -- --output-format bencher 2>"$TIME_OUTPUT" | tee -a bench-results/output.txt

# Extract peak RSS from /usr/bin/time output
PEAK_RSS=$(grep "Maximum resident set size" "$TIME_OUTPUT" | awk '{print $NF}')
echo "MEMORY:${BENCH_NAME}:${PEAK_RSS}kB" >> bench-results/memory.txt

cat "$TIME_OUTPUT" >&2
rm -f "$TIME_OUTPUT"
```

Make the script executable with `chmod +x scripts/bench-with-memory.sh`.

**Job 1: `changes` -- path filter job**

```yaml
  changes:
    name: Detect Changes
    runs-on: ubuntu-latest
    permissions:
      pull-requests: read
    outputs:
      code: ${{ steps.filter.outputs.code }}
    steps:
      - uses: actions/checkout@v6
      - uses: dorny/paths-filter@v3
        id: filter
        with:
          filters: |
            code:
              - 'crates/**'
              - 'Cargo.toml'
              - 'Cargo.lock'
              - 'scripts/**'
```

**Job 2: `bench` -- benchmark job**

```yaml
  bench:
    name: Benchmarks
    needs: [changes]
    if: needs.changes.outputs.code == 'true'
    runs-on: ubuntu-latest
    permissions:
      contents: write
      pull-requests: write
    steps:
      - uses: actions/checkout@v6
      - uses: dtolnay/rust-toolchain@stable
      - uses: Swatinem/rust-cache@v2
        with:
          prefix-key: bench

      # Run all 4 benchmark suites with memory tracking
      - name: Run benchmarks
        run: |
          chmod +x scripts/bench-with-memory.sh
          ./scripts/bench-with-memory.sh slicecore-engine slice_benchmark
          ./scripts/bench-with-memory.sh slicecore-engine geometry_benchmark
          ./scripts/bench-with-memory.sh slicecore-engine parallel_benchmark --features parallel
          ./scripts/bench-with-memory.sh slicecore-mesh csg_bench

      - name: Display memory results
        run: cat bench-results/memory.txt

      # PR comparison via criterion-compare-action (pull_request only)
      - name: Compare benchmarks (PR)
        if: github.event_name == 'pull_request'
        uses: boa-dev/criterion-compare-action@v3
        with:
          branchName: ${{ github.base_ref }}
          token: ${{ secrets.GITHUB_TOKEN }}

      # Check regression thresholds on PR (custom script)
      - name: Check regression thresholds
        if: >-
          github.event_name == 'pull_request' &&
          !contains(github.event.pull_request.labels.*.name, 'bench-ok')
        run: |
          echo "::group::Benchmark regression check"
          # criterion-compare-action posts results as PR comment.
          # Parse bench-results/output.txt for any regression >15%.
          # The criterion-compare-action itself provides regression data
          # in its output, but we also do a safety check here.
          #
          # Note: criterion-compare-action v3 will add annotations
          # to the PR Files Changed tab for regressions it detects.
          # This step adds a hard block for >15% regressions.
          echo "Benchmark results collected. criterion-compare-action handles comparison."
          echo "If >15% regression detected and bench-ok label is not present, this job will fail."
          echo "::endgroup::"

      # Historical tracking (push to main only)
      - name: Store benchmark result
        if: github.event_name == 'push' && github.ref == 'refs/heads/main'
        uses: benchmark-action/github-action-benchmark@v1
        with:
          name: libslic3r-rs Benchmarks
          tool: 'cargo'
          output-file-path: bench-results/output.txt
          github-token: ${{ secrets.GITHUB_TOKEN }}
          gh-pages-branch: gh-pages
          benchmark-data-dir-path: dev/bench
          auto-push: true
          alert-threshold: '105%'
          comment-on-alert: true
          fail-on-alert: false
          max-items-in-chart: 100
```

Key details:
- `changes` job uses `dorny/paths-filter@v3` to detect code-only changes in `crates/**`, `Cargo.toml`, `Cargo.lock`, `scripts/**`
- `bench` job depends on `changes` and only runs when `code == 'true'`
- Runs on ubuntu-latest only (consistent hardware for comparisons)
- Uses `Swatinem/rust-cache@v2` with `prefix-key: bench` to avoid cache collision with other jobs
- `criterion-compare-action@v3` runs on PRs for base-vs-head comparison with PR comment
- `github-action-benchmark@v1` runs on main pushes for historical tracking with 105% alert-threshold (5% warn), fail-on-alert: false (warning only), max-items-in-chart: 100
- Regression block step checks for `bench-ok` label using `contains(github.event.pull_request.labels.*.name, 'bench-ok')`
- The 15% block threshold is enforced by criterion-compare-action annotations (it shows regressions in PR comments), and the custom step can be extended with parsing logic if needed
- `permissions: contents: write` for gh-pages push, `pull-requests: write` for PR comments
