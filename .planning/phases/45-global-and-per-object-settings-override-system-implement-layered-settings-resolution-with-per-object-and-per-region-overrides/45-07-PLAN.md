Commands::Slice {

**Step 5: Validation and error handling**
Implement collect-all-errors pattern:
```rust
let mut errors: Vec<CliError> = Vec::new();
let mut warnings: Vec<String> = Vec::new();

// Validate all at once:
// - Missing files (mesh, override set, plate.toml)
// - Invalid override set names (fuzzy suggest)
// - Invalid field names in --object inline overrides
// - Object index out-of-bounds (with count and 1-indexing note)
// - Build plate size validation (objects fit within machine bed dimensions)
// - Basic bounding box collision detection between objects

if !errors.is_empty() {
    // Report all errors and exit with code 2
    for err in &errors { eprintln!("error: {err}"); }
    std::process::exit(2);
}

// --strict: warnings become errors
if strict && !warnings.is_empty() {
    for w in &warnings { eprintln!("error (--strict): {w}"); }
    std::process::exit(2);
}
// --force: suppress warnings
if !force {
    for w in &warnings { eprintln!("warning: {w}"); }
}
```
