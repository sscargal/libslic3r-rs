    ```rust
    #[test]
    fn sc1_native_plugin_loads_and_generates_infill() {
        // Prerequisites: build the native-zigzag-infill plugin first
        // The test script should run:
        //   cargo build --manifest-path plugins/examples/native-zigzag-infill/Cargo.toml
        // Then the .so will be at plugins/examples/native-zigzag-infill/target/debug/

        let plugin_dir = PathBuf::from(env!("CARGO_MANIFEST_DIR"))
            .join("../../plugins/examples/native-zigzag-infill");

        // Build the plugin first (in test setup)
        let status = std::process::Command::new("cargo")
            .args(["build", "--manifest-path"])
            .arg(plugin_dir.join("Cargo.toml"))
            .status()
            .expect("Failed to build native plugin");
        assert!(status.success(), "Native plugin build failed");

        // Load via registry
        let mut registry = PluginRegistry::new();
        let result = registry.discover_and_load(&plugin_dir);
        assert!(result.is_ok(), "Failed to discover plugin: {:?}", result.err());

        // Verify plugin is registered
        assert!(registry.has_infill_plugin("zigzag"));

        // List plugins
        let plugins = registry.list_infill_plugins();
        assert_eq!(plugins.len(), 1);
        assert_eq!(plugins[0].name, "zigzag");

        // Generate infill
        let plugin = registry.get_infill_plugin("zigzag").unwrap();
        let request = create_test_rectangle_request(100.0, 100.0, 0.2, 0, 0.2, 0.4);
        let result = plugin.generate(&request);
        assert!(result.is_ok(), "Plugin generate failed: {:?}", result.err());
        let infill = result.unwrap();
        assert!(!infill.lines.is_empty(), "Plugin produced no infill lines");
    }
    ```

    ```rust
    //! # slicecore-plugin-api
    //!
    //! Shared interface types for the slicecore plugin system.
    //!
    //! This crate defines the FFI-safe types and traits that form the contract
    //! between the slicecore host application and plugins. Both native (dynamic
    //! library) and WASM plugins implement the same logical interface.
    //!
    //! ## Architecture
    //!
    //! The plugin system uses a three-crate architecture:
    //!
    //! 1. **slicecore-plugin-api** (this crate) -- Shared types and traits
    //! 2. **slicecore-plugin** -- Host-side registry, loading, and lifecycle
    //! 3. **Plugin crates** -- Individual plugin implementations
    //!
    //! ## Creating a Native Plugin
    //!
    //! 1. Create a new crate with `crate-type = ["cdylib"]`
    //! 2. Depend on `slicecore-plugin-api`
    //! 3. Implement [`InfillPatternPlugin`] trait
    //! 4. Export the root module via `#[export_root_module]`
    //! 5. Create a `plugin.toml` manifest
    //! 6. Build with `cargo build`
    //!
    //! See `plugins/examples/native-zigzag-infill/` for a complete example.
    //!
    //! ## Creating a WASM Plugin
    //!
    //! 1. Create a new crate with `crate-type = ["cdylib"]`
    //! 2. Depend on `wit-bindgen`
    //! 3. Copy the WIT file from `crates/slicecore-plugin/wit/`
    //! 4. Use `wit_bindgen::generate!` to create guest bindings
    //! 5. Implement the `Guest` trait
    //! 6. Create a `plugin.toml` manifest
    //! 7. Build with `cargo component build` or `cargo build --target wasm32-wasip2`
    //!
    //! See `plugins/examples/wasm-spiral-infill/` for a complete example.
    ```
