    Sub-test 2b -- Crash isolation via fuel exhaustion with embedded WAT:
    ```rust
    #[test]
    #[cfg(feature = "wasm-plugins")]
    fn sc2b_wasm_plugin_fuel_exhaustion_does_not_crash_host() {
        // This test creates a minimal WASM component inline from WAT text
        // that enters an infinite loop in its generate() function.
        // With fuel set to 1, the wasmtime runtime will trap with a fuel
        // exhaustion error. The host MUST receive Result::Err, NOT crash.
        //
        // Approach: Use wasmtime directly (not through PluginRegistry) to
        // create a minimal component that always loops. This tests the
        // core isolation guarantee without needing cargo-component.

        use wasmtime::{Engine as WasmEngine, Config, Store};
        use wasmtime::component::Component;

        // Create a minimal WASM component that has an infinite loop.
        // The WAT defines a component implementing the infill-plugin world
        // with a generate function that loops forever.
        //
        // If writing a full Component Model WAT is too complex, use the
        // simpler approach: create a wasmtime Engine with fuel enabled,
        // load the spiral WASM plugin (if available) with max_cpu_fuel=1,
        // and call generate(). The fuel exhaustion will trap before any
        // meaningful computation completes.
        //
        // Fallback approach if no .wasm file exists:
        // Use wasmtime's core module (not component) to demonstrate fuel
        // exhaustion isolation, then document that full component-level
        // SC2 testing requires the built WASM plugin.

        let mut config = Config::new();
        config.consume_fuel(true);
        let engine = WasmEngine::new(&config).unwrap();

        // Minimal core WASM module with infinite loop
        let wat = r#"
            (module
                (func $loop (export "loop")
                    (loop $inf
                        (br $inf)
                    )
                )
            )
        "#;
        let module = wasmtime::Module::new(&engine, wat).unwrap();
        let mut store = Store::new(&engine, ());
        store.set_fuel(1).unwrap();
        let instance = wasmtime::Instance::new(&mut store, &module, &[]).unwrap();
        let loop_fn = instance.get_typed_func::<(), ()>(&mut store, "loop").unwrap();

        // Call the infinite loop -- fuel exhaustion MUST produce an error
        let result = loop_fn.call(&mut store, ());
        assert!(result.is_err(), "Expected fuel exhaustion error, got Ok");
        let err_msg = result.unwrap_err().to_string();
        assert!(
            err_msg.contains("fuel") || err_msg.contains("Fuel"),
            "Error should mention fuel exhaustion, got: {}",
            err_msg
        );

        // CRITICAL: If we reach this point, the host process survived.
        // The test passing IS the proof that crash isolation works.

        // Additionally, test with the full WASM plugin if available:
        let wasm_path = PathBuf::from(env!("CARGO_MANIFEST_DIR"))
            .join("../../plugins/examples/wasm-spiral-infill/target/wasm32-wasip2/debug/wasm_spiral_infill.wasm");
        if wasm_path.exists() {
            // Load with minimal fuel so generate() exhausts it
            let mut registry = PluginRegistry::new()
                .with_sandbox_config(SandboxConfig {
                    max_memory_bytes: 64 * 1024 * 1024,
                    max_cpu_fuel: 1, // Exhausts immediately on any computation
                });

            let plugin_dir = wasm_path.parent().unwrap().parent().unwrap()
                .parent().unwrap().parent().unwrap();
            // Note: loading may succeed (name/description queries use separate stores with
            // normal fuel). The generate() call should fail with fuel exhaustion.
            if let Ok(_) = registry.discover_and_load(plugin_dir) {
                if let Some(plugin) = registry.get_infill_plugin("spiral") {
                    let request = create_test_rectangle_request(10.0, 10.0, 0.2, 0, 0.2, 0.4);
                    let result = plugin.generate(&request);
                    assert!(result.is_err(), "Expected fuel exhaustion on generate(), got Ok");
                }
            }
        }
    }
    ```

    - The full component-level test runs when the WASM plugin is built (bonus coverage)

    ```rust
    //! # slicecore-plugin-api
    //!
    //! Shared interface types for the slicecore plugin system.
    //!
    //! This crate defines the FFI-safe types and traits that form the contract
    //! between the slicecore host application and plugins. Both native (dynamic
    //! library) and WASM plugins implement the same logical interface.
    //!
    //! ## Architecture
    //!
    //! The plugin system uses a three-crate architecture:
    //!
    //! 1. **slicecore-plugin-api** (this crate) -- Shared types and traits
    //! 2. **slicecore-plugin** -- Host-side registry, loading, and lifecycle
    //! 3. **Plugin crates** -- Individual plugin implementations
    //!
    //! ## Creating a Native Plugin
    //!
    //! 1. Create a new crate with `crate-type = ["cdylib"]`
    //! 2. Depend on `slicecore-plugin-api`
    //! 3. Implement [`InfillPatternPlugin`] trait
    //! 4. Export the root module via `#[export_root_module]`
    //! 5. Create a `plugin.toml` manifest
    //! 6. Build with `cargo build`
    //!
    //! See `plugins/examples/native-zigzag-infill/` for a complete example.
    //!
    //! ## Creating a WASM Plugin
    //!
    //! 1. Create a new crate with `crate-type = ["cdylib"]`
    //! 2. Depend on `wit-bindgen`
    //! 3. Copy the WIT file from `crates/slicecore-plugin/wit/`
    //! 4. Use `wit_bindgen::generate!` to create guest bindings
    //! 5. Implement the `Guest` trait
    //! 6. Create a `plugin.toml` manifest
    //! 7. Build with `cargo component build` or `cargo build --target wasm32-wasip2`
    //!
    //! See `plugins/examples/wasm-spiral-infill/` for a complete example.
    ```
