<objective>
Add WASM plugin loading to the slicecore-plugin crate using wasmtime Component Model, WIT interface definitions, and configurable sandboxing. Integrate WASM plugins into the existing PluginRegistry alongside native plugins.

Purpose: WASM plugins provide safety (can't crash host), portability (single binary runs everywhere), and security (sandboxed resource access). This satisfies SC2: "A WASM plugin loaded via wasmtime Component Model can provide a custom infill pattern, and a bug/crash in the WASM plugin does not crash or corrupt the host process."

Output: WIT interface file, wasmtime-based WASM plugin loader with sandboxing, and PluginRegistry integration for WASM plugins.
</objective>

    ```wit
    package slicecore:plugin@0.1.0;

    interface types {
        record point2 {
            x: s64,
            y: s64,
        }

        record infill-line {
            start: point2,
            end: point2,
        }

        record infill-request {
            boundary-points: list<point2>,
            boundary-lengths: list<u32>,
            density: f64,
            layer-index: u64,
            layer-z: f64,
            line-width: f64,
        }

        record infill-result {
            lines: list<infill-line>,
        }
    }

    world infill-plugin {
        use types.{infill-request, infill-result};

        export name: func() -> string;
        export description: func() -> string;
        export generate: func(request: infill-request) -> result<infill-result, string>;
    }
    ```

    Use `wasmtime::component::bindgen!` macro to generate host bindings from the WIT file:

    ```rust
    #[cfg(feature = "wasm-plugins")]
    mod wasm_impl {
        use wasmtime::component::{Component, Linker};
        use wasmtime::{Engine as WasmEngine, Store, Config};
        use wasmtime_wasi::WasiCtxBuilder;
        use std::path::Path;
        use crate::error::PluginSystemError;
        use crate::registry::InfillPluginAdapter;
        use crate::sandbox::SandboxConfig;
        use slicecore_plugin_api::types::{InfillRequest, InfillResult, FfiInfillLine};
        use abi_stable::std_types::RVec;

        wasmtime::component::bindgen!({
            world: "infill-plugin",
            path: "wit/slicecore-plugin.wit",
        });

        struct PluginState {
            wasi_ctx: wasmtime_wasi::WasiCtx,
            table: wasmtime::component::ResourceTable,
        }

        impl wasmtime_wasi::WasiView for PluginState {
            fn ctx(&mut self) -> &mut wasmtime_wasi::WasiCtx { &mut self.wasi_ctx }
            fn table(&mut self) -> &mut wasmtime::component::ResourceTable { &mut self.table }
        }

        pub struct WasmInfillPlugin {
            engine: WasmEngine,
            component: Component,
            sandbox_config: SandboxConfig,
            cached_name: String,
            cached_description: String,
        }

        impl WasmInfillPlugin {
            pub fn load(wasm_path: &Path, sandbox_config: SandboxConfig) -> Result<Self, PluginSystemError> {
                let mut config = Config::new();
                config.wasm_component_model(true);
                config.consume_fuel(true);

                let engine = WasmEngine::new(&config)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: wasm_path.to_path_buf(),
                        reason: format!("Failed to create WASM engine: {}", e),
                    })?;

                let component = Component::from_file(&engine, wasm_path)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: wasm_path.to_path_buf(),
                        reason: format!("Failed to load WASM component: {}", e),
                    })?;

                // Instantiate once to get name/description (cached)
                let (name, description) = Self::query_metadata(&engine, &component, &sandbox_config)?;

                Ok(Self {
                    engine,
                    component,
                    sandbox_config,
                    cached_name: name,
                    cached_description: description,
                })
            }

            fn query_metadata(
                engine: &WasmEngine,
                component: &Component,
                sandbox: &SandboxConfig,
            ) -> Result<(String, String), PluginSystemError> {
                // Create a temporary store with generous fuel to call name() and description().
                // These are lightweight calls that should never exhaust fuel.
                let wasi_ctx = WasiCtxBuilder::new().build();
                let state = PluginState {
                    wasi_ctx,
                    table: wasmtime::component::ResourceTable::new(),
                };
                let mut store = Store::new(engine, state);
                // Use generous fuel for metadata queries (not the sandbox limit,
                // which may be intentionally low for testing crash isolation).
                store.set_fuel(10_000_000).map_err(|e| PluginSystemError::LoadFailed {
                    path: std::path::PathBuf::new(),
                    reason: format!("Failed to set fuel: {}", e),
                })?;

                let mut linker = Linker::new(engine);
                wasmtime_wasi::add_to_linker_sync(&mut linker)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: std::path::PathBuf::new(),
                        reason: format!("Failed to link WASI: {}", e),
                    })?;

                // Instantiate the component using the generated bindgen types.
                // The exact API depends on the bindgen output, but typically:
                let (bindings, _instance) = InfillPlugin::instantiate(&mut store, component, &linker)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: std::path::PathBuf::new(),
                        reason: format!("Failed to instantiate WASM component: {}", e),
                    })?;

                let name = bindings.call_name(&mut store)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: std::path::PathBuf::new(),
                        reason: format!("Failed to call name(): {}", e),
                    })?;

                let description = bindings.call_description(&mut store)
                    .map_err(|e| PluginSystemError::LoadFailed {
                        path: std::path::PathBuf::new(),
                        reason: format!("Failed to call description(): {}", e),
                    })?;

                Ok((name, description))
            }

            fn create_store(&self) -> Result<Store<PluginState>, PluginSystemError> {
                let wasi_ctx = WasiCtxBuilder::new().build();
                let state = PluginState {
                    wasi_ctx,
                    table: wasmtime::component::ResourceTable::new(),
                };
                let mut store = Store::new(&self.engine, state);
                store.set_fuel(self.sandbox_config.max_cpu_fuel)
                    .map_err(|e| PluginSystemError::ExecutionFailed {
                        plugin: self.cached_name.clone(),
                        message: e.to_string(),
                    })?;
                Ok(store)
            }
        }
    }
    ```

    Implement `InfillPluginAdapter` for `WasmInfillPlugin`:
    - `name()` returns cached_name.clone()
    - `description()` returns cached_description.clone()
    - `generate(request)`:
      1. Create a fresh Store with fuel limits via create_store()
      2. Instantiate the component (new linker + instantiate for each call)
      3. Convert InfillRequest to WIT-generated request type
      4. Call the generated `generate` function
      5. Convert WIT result back to InfillResult
      6. If the WASM plugin traps, return PluginSystemError::ExecutionFailed (NOT a panic)
    - `plugin_type()` returns PluginType::Wasm

    ```rust
    /// Configuration for WASM plugin sandboxing.
    #[derive(Debug, Clone, Serialize, Deserialize)]
    pub struct SandboxConfig {
        /// Maximum memory in bytes for the WASM plugin (default: 64 MiB).
        pub max_memory_bytes: usize,
        /// Maximum CPU fuel units (default: 1_000_000, roughly 1M wasm instructions).
        pub max_cpu_fuel: u64,
    }

    impl Default for SandboxConfig {
        fn default() -> Self {
            Self {
                max_memory_bytes: 64 * 1024 * 1024, // 64 MiB
                max_cpu_fuel: 10_000_000,            // ~10M instructions
            }
        }
    }
    ```
