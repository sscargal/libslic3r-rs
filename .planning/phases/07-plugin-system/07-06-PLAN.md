    **src/lib.rs:**
    Use wit-bindgen to generate guest bindings and implement the infill-plugin world:

    ```rust
    wit_bindgen::generate!({
        world: "infill-plugin",
        path: "wit/slicecore-plugin.wit",
    });

    struct SpiralInfillPlugin;

    impl Guest for SpiralInfillPlugin {
        fn name() -> String {
            "spiral".to_string()
        }

        fn description() -> String {
            "Spiral infill: concentric inward-spiraling paths".to_string()
        }

        fn generate(request: InfillRequest) -> Result<InfillResult, String> {
            // 1. Reconstruct bounding box from boundary_points
            // 2. Compute center point
            // 3. Generate a spiral from the bounding box inward
            //    - Start at outer boundary, spiral inward with spacing based on density
            //    - Each revolution decreases radius by line_width/density
            //    - Approximate spiral as line segments (angular step ~5-10 degrees)
            // 4. Return infill lines

            let (min_x, min_y, max_x, max_y) = compute_bbox(&request.boundary_points);
            let center_x = (min_x + max_x) / 2;
            let center_y = (min_y + max_y) / 2;

            let width = (max_x - min_x) as f64;
            let height = (max_y - min_y) as f64;
            let max_radius = width.min(height) / 2.0;

            // Spacing between spiral arms
            let spacing = request.line_width / request.density;
            let spacing_coord = (spacing * 1_000_000.0) as i64; // mm to coord

            let mut lines = Vec::new();
            let num_revolutions = (max_radius / (spacing * 1_000_000.0)).ceil() as usize;
            let steps_per_rev = 36; // 10-degree steps

            let mut prev_x = center_x + (max_radius as i64);
            let mut prev_y = center_y;

            for rev in 0..num_revolutions {
                for step in 1..=steps_per_rev {
                    let total_step = rev * steps_per_rev + step;
                    let angle = (total_step as f64) * (2.0 * std::f64::consts::PI / steps_per_rev as f64);
                    let radius = max_radius - (total_step as f64 * spacing * 1_000_000.0 / steps_per_rev as f64);
                    if radius <= 0.0 { break; }
                    let x = center_x + (radius * angle.cos()) as i64;
                    let y = center_y + (radius * angle.sin()) as i64;
                    lines.push(InfillLine {
                        start: Point2 { x: prev_x, y: prev_y },
                        end: Point2 { x, y },
                    });
                    prev_x = x;
                    prev_y = y;
                }
            }

            Ok(InfillResult { lines })
        }
    }

    export!(SpiralInfillPlugin);
    ```

    Note: The exact spiral algorithm details may need adjustment once we see the actual WIT-generated type names. The key requirement is that the plugin produces non-empty infill lines for a given boundary.

    Keep the algorithm intentionally different from the native zigzag example (spiral vs zigzag) so both examples are distinct and demonstrate different infill patterns.
