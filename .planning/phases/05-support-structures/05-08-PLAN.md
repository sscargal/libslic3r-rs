**SC1: Automatic support generation correctly identifies overhangs and generates traditional grid/line support structures.**

```rust
#[test]
fn sc1_auto_support_identifies_overhangs_and_generates_traditional_support() {
    let mesh = overhang_block();
    let mut config = PrintConfig::default();
    config.support.enabled = true;
    config.support.support_type = SupportType::Traditional;
    config.support.overhang_angle = 45.0;

    let engine = Engine::new(config);
    let result = engine.slice(&mesh).unwrap();
    let gcode = String::from_utf8_lossy(&result.gcode);

    // G-code must contain support toolpath comments.
    assert!(gcode.contains("TYPE:Support"), "G-code should contain support toolpaths");

    // G-code must be valid.
    let validation = slicecore_gcode_io::validate_gcode(&gcode);
    assert!(validation.errors.is_empty(), "G-code should be valid: {:?}", validation.errors);

    // Support should only appear on layers where overhang exists (not on all layers).
    // Count layers with TYPE:Support — should be less than total layers.
    let support_layers: usize = gcode.lines()
        .filter(|l| l.contains("TYPE:Support"))
        .count();
    assert!(support_layers > 0, "Should have support on some layers");
    assert!(support_layers < result.layer_count, "Should not have support on every layer");
}
```

**SC2: Tree supports use less material than traditional supports.**

```rust
#[test]
fn sc2_tree_supports_use_less_material_than_traditional() {
    let mesh = overhang_with_small_features();

    // Slice with traditional support.
    let mut trad_config = PrintConfig::default();
    trad_config.support.enabled = true;
    trad_config.support.support_type = SupportType::Traditional;
    let trad_result = Engine::new(trad_config).slice(&mesh).unwrap();

    // Slice with tree support.
    let mut tree_config = PrintConfig::default();
    tree_config.support.enabled = true;
    tree_config.support.support_type = SupportType::Tree;
    let tree_result = Engine::new(tree_config).slice(&mesh).unwrap();

    // Extract total E-axis values (filament usage proxy).
    let trad_e = extract_total_extrusion(&trad_result.gcode);
    let tree_e = extract_total_extrusion(&tree_result.gcode);

    // Tree should use less filament than traditional.
    assert!(tree_e < trad_e,
        "Tree supports ({:.2}mm) should use less material than traditional ({:.2}mm)",
        tree_e, trad_e);
}
```

**Helper function:**

```rust
fn extract_total_extrusion(gcode: &[u8]) -> f64 {
    let text = String::from_utf8_lossy(gcode);
    text.lines()
        .filter_map(|line| {
            if line.starts_with("G1") {
                line.split_whitespace()
                    .find(|w| w.starts_with('E'))
                    .and_then(|e| e[1..].parse::<f64>().ok())
            } else {
                None
            }
        })
        .sum()
}
```
