use clap::Args;
use slicecore_config_schema::{Tier, SettingCategory};
use slicecore_engine::setting_registry;

#[derive(Args)]
pub struct SchemaArgs {
    /// Output format: json-schema or json
    #[arg(long, default_value = "json-schema")]
    format: SchemaFormat,

    /// Filter by maximum tier: simple, intermediate, advanced, developer
    #[arg(long)]
    tier: Option<TierFilter>,

    /// Filter by category: speed, quality, cooling, retraction, support, infill, etc.
    #[arg(long)]
    category: Option<String>,

    /// Full-text search across key, display name, description, and tags
    #[arg(long)]
    search: Option<String>,
}

#[derive(Clone, clap::ValueEnum)]
enum SchemaFormat {
    JsonSchema,
    Json,
}

#[derive(Clone, clap::ValueEnum)]
enum TierFilter {
    Simple,
    Intermediate,
    Advanced,
    Developer,
}

impl TierFilter {
    fn to_tier(&self) -> Tier {
        match self {
            TierFilter::Simple => Tier::Simple,
            TierFilter::Intermediate => Tier::Intermediate,
            TierFilter::Advanced => Tier::Advanced,
            TierFilter::Developer => Tier::Developer,
        }
    }
}
```

Define validation types:

/// Severity level for a validation issue.
#[derive(Debug, Clone, Copy, PartialEq, Eq, Serialize, Deserialize)]
pub enum ValidationSeverity {
    /// Informational: depends_on condition not met
    Info,
    /// Suspicious but not necessarily dangerous
    Warning,
    /// Dangerous value that should prevent slicing
    Error,
}

/// A single validation issue found in the config.
#[derive(Debug, Clone, Serialize, Deserialize)]
pub struct ValidationIssue {
    /// Config field key that triggered the issue
    pub key: String,
    /// Human-readable description of the problem
    pub message: String,
    /// Severity level
    pub severity: ValidationSeverity,
}
```

Implement on SettingRegistry:
```rust
pub fn validate_config(&self, config_json: &serde_json::Value) -> Vec<ValidationIssue>
```

Logic:
- For each SettingDefinition in the registry:
  - Resolve the field value from config_json using dotted key path traversal
  - If value is missing and field is not optional -> Warning
  - For each Constraint::Range { min, max }:
    - If value is a number and value < min or value > max -> Error with message like "speed.perimeter value 5000.0 exceeds maximum 1000.0"
    - If value is very close to min or max boundary -> Warning
  - For each Constraint::DependsOn { key, condition }:
    - Resolve the dependency key's value
    - If dependency condition is not met (e.g., "support.enable" is false) but this field has non-default value -> Info level: "speed.support is configured but support.enable is false"
  - For deprecated fields: if value differs from default -> Warning with deprecation message
